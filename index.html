An Ansible Playbook

 
---
- hosts: all
  gather_facts: False
  connection: ssh
  vars:
    bootstrap: false
  tasks:
    - raw: aptitude install -y python-minimal
      when: "{{ bootstrap }}"
    - apt: name=python-zmq
      when: "{{ bootstrap }}"
    - apt: name=python-keyczar
      when: "{{ bootstrap }}"

Running a playbook

ansible-playbook -i inventory bootstrap.yml -e bootstrap=true

That's It!
